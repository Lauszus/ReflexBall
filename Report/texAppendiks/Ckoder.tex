//keeps program running with while(true)-loop. On mouse click zooms in on mouse (x,y)-coordinates
public static void zoom() {
	while(true) {
		if(StdDraw.mousePressed()) {
			center[0] = StdDraw.mouseX();
			center[1] = StdDraw.mouseY();
			sideLength /= 2.0;
			setupDraw();
			drawMandelbrot();
		}
	}
}
\end{lstlisting}
